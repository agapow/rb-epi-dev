                                        def [] *args
                                                hsh = OrderedHash.new
                                                if Hash === args[0]
                                                        hsh.replace args[0]
                                                elsif (args.size % 2) != 0
                                                        raise ArgumentError, "odd number of elements for Hash"
                                                else
                                                        0.step(args.size - 1, 2) do |a|
                                                                b = a + 1
                                                                hsh[args[a]] = args[b]
                                                        end
                                                end
                                                hsh
                                        end
