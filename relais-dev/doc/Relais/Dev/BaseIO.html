def initialize(io_or_path, mode)
  # TODO: what if it's a Pathname?

  # if passed a filepath, open it
  is_path = (io_or_path.class == String)
  if is_path
    @hndl = File.open(io_or_path, mode)
    @file_open = true
  else
    @hndl = io_or_path
    @file_open = false
  end
  prepare()
end
